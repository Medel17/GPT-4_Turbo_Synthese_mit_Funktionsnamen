public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        char c = input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += (int)c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    check %= 256;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char checksumChar = (char)check;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    checksumChar = (char)(((int)checksumChar % 65) + 65);
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + checksumChar;
    // ### HOLE END
}
