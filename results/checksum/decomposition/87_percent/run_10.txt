public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        char c = input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += (int)c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    int csum = check % 256;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char checksumChar = (char)(csum);
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (checksumChar <= ' ') {
        checksumChar += 'A' - ' ';
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + checksumChar;
    // ### HOLE END
}
