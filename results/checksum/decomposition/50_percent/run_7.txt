public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        int c = (int) input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    check %= 256;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char resultChar = (char) check;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (resultChar == '!') {
        resultChar = 'A';
    } else if (resultChar == '?') {
        resultChar = '_';
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    String output = "Check sum is " + resultChar;
    // ### HOLE END
    return output;
}
