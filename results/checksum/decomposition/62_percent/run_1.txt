public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        int c = input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    while (check > 127) {
        check -= 127;
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char resultChar = (char)check;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (!Character.isLetterOrDigit(resultChar)) {
        resultChar = resultChar == '!' ? 'A' : '_';
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + resultChar;
    // ### HOLE END
}
