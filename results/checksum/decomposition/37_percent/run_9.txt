public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        int c = (int) input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    int modulo = check % 256;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char finalChar = (char) modulo;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (finalChar < ' ' || finalChar > '~') {
        finalChar = (finalChar % 94) + ' ';
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + finalChar;
    // ### HOLE END
}
