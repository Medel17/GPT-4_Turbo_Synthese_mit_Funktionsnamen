public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        int c = (int) input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    while (check >= 128) {
        check %= 128;
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char result = (char) check;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (result < '0' || result > '?') {
        result += '?';
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + result;
    // ### HOLE END
}
