public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        int c = input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += c;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    check %= 256;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char checkChar = (char) check;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (checkChar == '0') checkChar = 'P';
    else if (checkChar == '!') checkChar = 'A';
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + checkChar;
    // ### HOLE END
}
