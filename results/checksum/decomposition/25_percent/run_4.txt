public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        char c = input.charAt(i);
        int cval = (int)c;
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check += cval;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    check %= 256;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char finalChar = (char) check;
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    if (finalChar < ' ' || finalChar > '~') {
        finalChar = 'P';
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + finalChar;
    // ### HOLE END
}
