public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        char c = input.charAt(i);
        int numericValue = (int)c;
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        check = check + numericValue;
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    while (check > 127) {
        check -= 127;
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    char finalChar = (char)check; 
    // ### HOLE END
    // ### HOLE BEGIN clamp5
    switch (finalChar) {
        case '!':
            finalChar = '!';
            break;
        case '0':
            finalChar = '0';
            break;
        case '?':
            finalChar = '?';
            break;
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp6
    return "Check sum is " + finalChar;
    // ### HOLE END
}
