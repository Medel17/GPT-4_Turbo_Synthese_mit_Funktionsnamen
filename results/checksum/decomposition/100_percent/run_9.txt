public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i); // hole1
        check += c; // hole2
    }
    check %= 256; // hole3
    char finalChar = (char) check; // hole4
    if (finalChar < ' ' || finalChar > '~') { // hole5
        finalChar = (char) ('A' + (finalChar % 26)); // hole5
    }
    return "Check sum is " + finalChar; // hole6
}
