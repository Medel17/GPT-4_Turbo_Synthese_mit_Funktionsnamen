public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### Missing section
        char c = input.charAt(i);
        if (i % 2 == 0) { 
            check += c;    // Add char values for even indexes
        } else { 
            check -= c;    // Subtract char values for odd indexes
        }
        // ### End of Missing section
    }
    check &= 0xFF;  // Reduce to a single byte value
    if (check >= 32 && check <= 126) { // Printable ASCII characters range
        return "Check sum is " + (char)check; // Return as character
    } else {
        return "Check sum is "; // Return empty if not a valid ASCII printable character
    }
}
