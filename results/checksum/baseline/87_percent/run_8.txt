public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check = (check + input.charAt(i)) % 256;
    }
    if (check == 0 && input.length() > 0) {
        check = 256;
    }
    return "Check sum is " + (input.length() == 0 ? "" : (char) check);
}
