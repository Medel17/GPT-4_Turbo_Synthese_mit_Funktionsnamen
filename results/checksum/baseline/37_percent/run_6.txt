public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= input.charAt(i); // Perform bitwise XOR on each character's ASCII value
    }
    check = (check % 94) + 32; // Normalize to printable ASCII characters range
    String result = "Check sum is " + (char)check;
    return result;
}
