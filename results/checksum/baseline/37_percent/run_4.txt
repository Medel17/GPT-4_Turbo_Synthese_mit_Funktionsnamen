public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check += input.charAt(i);
    }
    check = (check % 95) + 32; // Modulo 95 to cycle through printable characters, offset by 32
    char checksumCharacter = (char) (check);
    return "Check sum is " + checksumCharacter;
}
