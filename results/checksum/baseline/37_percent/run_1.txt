public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= (int) input.charAt(i);
        check += (check << 1) + (check << 3) + (i | 1); // Optimized calculation through shifts and OR
    }
    return "Check sum is " + (char)(check % 256);
}
