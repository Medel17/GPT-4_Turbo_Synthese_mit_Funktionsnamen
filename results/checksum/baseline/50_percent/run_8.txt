public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= (int) input.charAt(i);
    }
    char checksumChar = (char) (check % 128); // use modulo 128 to ensure it is a valid ASCII character
    return "Check sum is " + checksumChar;
}
