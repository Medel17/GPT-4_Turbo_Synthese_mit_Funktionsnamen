public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= input.charAt(i); // Exclusive OR each character's ASCII value with 'check'
    }
    if (input.isEmpty()) {
        return "Check sum is ";
    } else {
        return "Check sum is " + (char) check;
    }
}
