public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= input.charAt(i);
        // Perform left circular shift of 1 position.
        check = (check << 1) | (check >>> (Integer.SIZE - 1));
    }
    return "Check sum is " + (char) check;
}
