public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check += input.charAt(i); // Sum up ASCII values of characters
        check = check % 256; // Ensure it stays within one-byte limit
    }

    char checksumChar = (char) check; // Convert final check value to corresponding ASCII character
    return "Check sum is " + checksumChar;
}
