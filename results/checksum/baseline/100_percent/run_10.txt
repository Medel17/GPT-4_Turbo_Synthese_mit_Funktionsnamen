public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= input.charAt(i); // Bitwise XOR each character's ASCII value with `check`
    }

    char result = (char) (check % 256); // Ensure the result is a valid ASCII character
    return "Check sum is " + result;
}
