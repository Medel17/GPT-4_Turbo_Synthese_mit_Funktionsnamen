public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        check ^= c;  // XOR each character of the string into 'check'
    }
    
    // Modulo 128 to keep the character within ASCII printable range
    check = (check % 128);

    // Check if the resulted ASCII is printable, if not, just print blank
    if (check < 32 || check > 126) {
        check = ' ';  // if non-printable, default to space character
    }
    
    return "Check sum is " + (char)check;
}
