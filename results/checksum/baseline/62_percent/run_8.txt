public static String Checksum(String input) {
    int check = 0;
    for (int i = 0; i < input.length(); i++) {
        check ^= input.charAt(i); // XOR each character with the accumulated checksum
    }

    // Modulo 128 to get ASCII range, assuming the test cases confirm with this modulo
    char result = (char) (check % 128);
    return "Check sum is " + result;
}
