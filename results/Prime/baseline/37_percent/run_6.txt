public static boolean Prime(int n) {
    if (n <= 1) return false; // prime numbers are greater than 1
    if (n <= 3) return true; // 2 and 3 are prime numbers
    if (n % 2 == 0 || n % 3 == 0) return false; // multiples of 2 and 3 are not primes
    
    int r = (int)Math.sqrt(n); // check up to the square root of n
    int f = 5;
    while (f <= r) {
        if (n % f == 0 || n % (f + 2) == 0) return false;
        f += 6;
    }
    return true;
}
