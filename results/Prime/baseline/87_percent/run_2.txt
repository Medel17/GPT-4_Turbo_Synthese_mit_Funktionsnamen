public static boolean Prime(int n) {
    if (n <= 1) { // 1 or less are not primes
        return false;
    }
    if (n <= 3) { // 2 and 3 are prime numbers
        return true;
    }
    if (n % 2 == 0 || n % 3 == 0) { // multiples of 2 and 3 are not primes if greater than 2 or 3
        return false;
    }

    int r = (int) Math.sqrt(n);
    int f = 5;
    while (f <= r) {
        if (n % f == 0 || n % (f + 2) == 0) {
            return false;
        }
        f += 6;
    }
    return true;
}
