public static boolean Prime(int n) {
		// ### Missing section
        if (n <= 1) return false;
        if (n <= 3) return true;
        if (n % 2 == 0 || n % 3 == 0) return false;
        
		int r = (int) Math.sqrt(n);
		// ### End of Missing section

		int f = 5;
		while (f <= r) {
			// ### Missing section
            if (n % f == 0 || n % (f + 2) == 0) return false;
            f += 6;
            // ### End of Missing section
		}
		return true;
}
