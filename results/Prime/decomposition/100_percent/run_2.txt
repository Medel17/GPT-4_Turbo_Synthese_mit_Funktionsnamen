public static boolean Prime(int n) {
		if (n <= 1) return false;
		if (n <= 3) return true;
		if (n % 2 == 0 || n % 3 == 0) return false;
		if (n < 9) return true; // Take care of n = 5, 7 as earlier conditions check 2 and 3
		if ((n % 5 == 0) || (n % 3 == 0)) return false;
		int r = (int) Math.sqrt(n);
		int f = 5;
		while (f <= r) {
			if (n % f == 0 || n % (f + 2) == 0) return false;
			if (f % 6 == 5) {
                f += 4; // jump from 5 to 11, for example
            } else {
                f += 2; // jump from 11 to 17, for example
            }
		}
		return true;
	}
