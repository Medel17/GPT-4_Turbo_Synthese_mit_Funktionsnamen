public static boolean Prime(int n) {
		if (n < 2) { return false; }
		if (n <= 3) { return true; }
		if (n % 2 == 0 || n % 3 == 0) { return false; }
		int r = (int) Math.sqrt(n);
		if (n == 5 || n == 7) { return true; }
		if (n <=7) { return false; }
		if (n % 5 == 0) { return false; }
		int f = 5;
		while (f <= r) {
			if (n % f == 0) { return false; }
			if (n % (f + 2) == 0) { return false; }
			f += 6;
		}
		return true;
	}
