public static boolean Prime(int n) {
		if (n <= 1) {
			return false;
		}
		if (n == 2) {
			return true;
		}
		if (n % 2 == 0) {
			return false;
		}
		if (n == 3) {
			return true;
		}
		if (n % 3 == 0) {
			return false;
		}
		int r = (int)Math.sqrt(n);
		int f = 5;
		while (f <= r) {
			if (n % f == 0) {
				return false;
			}
			if (n % (f + 2) == 0) {
				return false;
			}
			f += 6;
		}
		return true;
	}
