public static boolean Prime(int n) {
    if (n <= 1) return false; // Hole 1

    if (n <= 3) return true; // Hole 2

    if (n % 2 == 0 || n % 3 == 0) return false; // Hole 3

    if (n == 5 || n == 6) return true; // Hole 4

    if (n % 5 == 0 || n % 3 == 0) return false; // Hole 5

    int r = (int) Math.sqrt(n); // Hole 6

    int f = 5;
    while (f <= r) {
        if (n % f == 0) return false; // Hole 7
        if (n % (f + 2) == 0) return false; // Hole 8
        f += 6; // Hole 9
    }
    return true;
}
