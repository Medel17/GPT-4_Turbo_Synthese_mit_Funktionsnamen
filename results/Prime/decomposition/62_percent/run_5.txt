public static boolean Prime(int n) {
		if (n <= 1) {
			return false;
		}
		if (n <= 3) {
			return true;
		}
		if (n % 2 == 0 || n % 3 == 0) {
			return false;
		}
		if (n == 5) {
		    return true;
		}
		int r = (int)Math.sqrt(n);
		int f = 5;
		while (f <= r) {
			if (n % f == 0 || n % (f + 2) == 0) {
				return false;
			}
			f += 6;
		}
		return true;
	}
