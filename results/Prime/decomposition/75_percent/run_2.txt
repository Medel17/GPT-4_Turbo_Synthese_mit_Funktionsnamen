public static boolean Prime(int n) {
		if(n < 2) return false;
		if(n == 2 || n == 3) return true;
		if(n % 2 == 0 || n % 3 == 0) return false;
		for (int i = 5; i * i <= n; i += 6) {
			if (n % i == 0 || n % (i + 2) == 0) return false;
		}
		int f = 5;
		while (f * f <= n) {
			if (n % f == 0 || n % (f + 2) == 0) return false;
			f += 6;
		}
		return true;
	}
