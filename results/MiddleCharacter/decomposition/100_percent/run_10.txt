public static String MiddleCharacter(String string) {
	// ### HOLE BEGIN clamp1
	if (string == null || string.isEmpty()) {
		return "";
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	int n = string.length();
	if (n % 2 == 0) {
		return string.substring(n/2 - 1, n/2 + 1);
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	return string.substring(n/2, n/2 + 1);
	// ### HOLE END
}
