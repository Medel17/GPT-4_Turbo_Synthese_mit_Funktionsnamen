public static String MiddleCharacter(String string) {
    if (string == null || string.length() == 0) {
        return "";
    }

    int length = string.length();
    if (length % 2 == 0) {
        return string.substring(length / 2 - 1, length / 2 + 1);
    } else {
        return String.valueOf(string.charAt(length / 2));
    }
}
