public static String MiddleCharacter(String string) {
	// ### HOLE BEGIN clamp1
	if (string == null || string.isEmpty()) {
		return "";
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	int len = string.length();
	if (len % 2 == 0) {
		return string.substring(len / 2 - 1, len / 2 + 1);
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	return String.valueOf(string.charAt(len / 2));
	// ### HOLE END
}
