public static String MiddleCharacter(String string) {
    if (string == null || string.isEmpty()) {
        return "";
    }

    int len = string.length();
    int half = len / 2;

    if (len % 2 == 0) {
        return string.substring(half - 1, half + 1);
    } else {
        return String.valueOf(string.charAt(half));
    }
}
