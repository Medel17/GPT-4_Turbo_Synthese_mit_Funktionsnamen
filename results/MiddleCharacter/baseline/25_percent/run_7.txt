public static String MiddleCharacter(String string) {
    if (string == null || string.isEmpty()) {
        return "";
    }
    int len = string.length();
    int midIndex = len / 2;
    if (len % 2 == 0) {
        // Even length strings
        return string.substring(midIndex - 1, midIndex + 1);
    } else {
        // Odd length strings
        return string.substring(midIndex, midIndex + 1);
    }
}
