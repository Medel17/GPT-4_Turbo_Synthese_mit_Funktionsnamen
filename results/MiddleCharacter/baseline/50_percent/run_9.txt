public static String MiddleCharacter(String string) {
    int len = string.length();
    if (len == 0) {
        return "";
    }
    int midIndex = len / 2;
    if (len % 2 == 0) {
        return string.substring(midIndex - 1, midIndex + 1);
    } else {
        return String.valueOf(string.charAt(midIndex));
    }
}
