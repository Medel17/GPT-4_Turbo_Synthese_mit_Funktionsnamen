public static int basement(int[] arr) {
    int sum = 0;
    for (int i = 0; i < arr.length; i++) {
        sum += arr[i];  // Add the current element of the array to sum
        if (sum < 0) {
            return i;  // If the cumulative sum is negative, return the current index
        }
    }
    return -1;  // If the sum never goes negative, return -1
}
