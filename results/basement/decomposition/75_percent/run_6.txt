public static int basement(int[] arr) {
    int sum = 0;
    for (int i = 0; i < arr.length; i++) {
        // ### HOLE BEGIN clamp1
        sum += arr[i];
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        if (sum < 0) return i;
        // ### HOLE END
    }
    return -1;
}
