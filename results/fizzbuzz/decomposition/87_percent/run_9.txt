public static String fizzbuzz(int x) {
    String result = "";
    // ### HOLE BEGIN fizz
    if (x % 3 == 0) {
        result += "Fizz";
    }
    // ### HOLE END
    // ### HOLE BEGIN buzz
    if (x % 5 == 0) {
        result += "Buzz";
    }
    // ### HOLE END
    // ### HOLE BEGIN number
    if (x % 3 != 0 && x % 5 != 0) {
        result += Integer.toString(x);
    }
    // ### HOLE END
    return result;
}
