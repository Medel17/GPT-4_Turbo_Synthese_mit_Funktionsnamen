public static String fizzbuzz(int x) {
    String result = "";
    // ### Missing section
    if (x % 3 == 0) {
        result += "Fizz";
    }
    if (x % 5 == 0) {
        result += "Buzz";
    }
    if (result.isEmpty()) {
        result = String.valueOf(x);
    }
    // ### End of Missing section
    return result;
}
