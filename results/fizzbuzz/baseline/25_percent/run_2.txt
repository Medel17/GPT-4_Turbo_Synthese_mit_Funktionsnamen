public static String fizzbuzz(int x) {
    String result = "";
    
    if (x % 15 == 0) {
        result = "FizzBuzz";
    } else if (x % 3 == 0) {
        result = "Fizz";
    } else if (x % 5 == 0) {
        result = "Buzz";
    } else {
        result = Integer.toString(x);
    }
    
    return result;
}
