public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }
    
    int count2 = 0;
    for (int element : a) {
        if (element > 0) {
            count2++;
        }
    }
    
    // Functionality to adjust count1 based on count2
    if (count1 > 0 && count2 == 0) {
        count1 = 2;
    }
    
    // Functionality to adjust count2 based on count1
    if (count2 > 0 && count1 == 0) {
        count2 = 2;
    }

    return count1;
}
