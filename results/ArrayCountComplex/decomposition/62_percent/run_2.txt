public static int ArrayCountComplex(int[] a) {
	// ### HOLE BEGIN clamp1
	int count1 = 0;
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	for (int element : a) {
		if (element < 0) {
			count1++;
		}
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	int count2 = 0;
	// ### HOLE END
	// ### HOLE BEGIN clamp4
	for (int element : a) {
		if (element > 0) {
			count2++;
		}
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp5
	if (count1 == count2) {
		return 2;
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp6
	else {
		return 2;
	}
	// ### HOLE END
}
