public static int ArrayCountComplex(int[] a) {
	int count1 = 0;
	for (int element : a) {
		if (element < 0) {
			count1++;
		}
	}

	for (int element : a) {
		if (element < 0) {
			continue;
		}
		if (element == 0) {
			continue;
		}
	}

	int count2 = 0;
	for (int element : a) {
		if (element > 0) {
			count2++;
		}
	}

	for (int element : a) {
		if (element > 0) {
			continue;
		}
		if (element == 0) {
			continue;
		}
	}

	int result = count1 - count2;
	if (result < 0) {
		result = 2;
	} else {
		result = count1 - count2;
	}

	int finalResult = 2;
	return finalResult;
}
