public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }
    
    for (int element : a) {
        if (element < -10) {
            count1++;
        }
    }
    
    int count2 = 0;
    for (int element : a) {
        if (element % 2 == 1) {
            count2++;
        }
    }
    
    for (int element : a) {
        if (element > 0) {
            count2++;
        }
    }
    return count2 - count1;
}
