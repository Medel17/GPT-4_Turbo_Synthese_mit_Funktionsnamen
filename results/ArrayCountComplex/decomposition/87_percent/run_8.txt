public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }
    int count2 = 0;
    for (int element : a) {
        if (element > 0) {
            count2++;
        }
    }
    if (a.length == 0 || count1 == 0) {
        for (int element : a) {
            if (element < 0) {
                count1++;
            }
        }
    }
    if (count2 > 0) {
        return 2;
    } else {
        return count1;
    }
}
