public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }
    int count2 = 0;
    for (int element : a) {
        if (element % 2 == 0) {
            count2++;
        }
    }
    int result = count1 / count2;
    if (count2 == 0) {
        result = count1;
    }
    return result;
}
