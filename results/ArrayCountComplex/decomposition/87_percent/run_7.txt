public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    int count2 = 0;
    for (int element : a) {
        if (element < 0) {
            count1 += 1;
        }
        if (element % 2 != 0) {
            count2 += 1;
        }
    }
    return count1 > count2 ? count1 - count2 : count2 - count1;
}
