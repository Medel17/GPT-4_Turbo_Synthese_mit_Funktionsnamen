public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }
    int count2 = 0;
    for (int element : a) {
        if (element > 0) {
            count2++;
        }
    }
    int totalCount = 0;
    totalCount = count1;
    totalCount = totalCount + count2;
    return totalCount;
}
