public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }
    int count2 = 0;
    for (int element : a) {
        if (element > 0) {
            count2++;
        }
    }
    int result = 0;
    if (count1 >= count2) {
        result = count1;
    } else {
        result = count2;
    }
    return result;
}
