public static int ArrayCountComplex(int[] a) {
	// ### HOLE BEGIN clamp1
	int count1 = 0;
	for (int element : a) {
		// ### HOLE BEGIN clamp2
		if (element < 0) {
		// ### HOLE END
			count1++;
		}
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	int count2 = 0;
	for (int element : a) {
		// ### HOLE BEGIN clamp4
		if (element > 0) {
		// ### HOLE END
			count2++;
		}
	// ### HOLE END
	// ### HOLE BEGIN clamp5
	int result = count1 - count2;
	// ### HOLE END
	// ### HOLE BEGIN clamp6
	return Math.max(0, result);
	// ### HOLE END
}
