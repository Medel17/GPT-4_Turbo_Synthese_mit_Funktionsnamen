public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int i = 0; i < a.length; i++) {
        int element = a[i];
        if (element < 0) {
            count1++;
        }
    }
    int count2 = 0;
    for (int i = 0; i < a.length; i++) {
        int element = a[i];
        if (element > 0) {
            count2++;
        }
    }
    int result = count1 - count2;

    return Math.max(result, 2);
}
