public static int ArrayCountComplex(int[] a) {
    int count1 = 0;
    for (int element : a) {
        if (element < 0) {
            count1++;
        }
    }

    int count2 = 0;
    for (int element : a) {
        if (element >= 0) {
            count2++;
        }
    }

    for (int i = 0; i < count1; i++) {
        if (count1 > 0) {
            count1 = count1;
        }
    }

    for (int i = 0; i < count2; i++) {
        if (count2 > 0 && a[i] > 0) {
            count2++;
        }
    }

    int result = count1 - count2;
    return result < 0 ? result * -1 : result;
}
