public static int ArrayCountComplex(int[] a) {
	int count1 = 0;
	for (int element : a) {
		if (element < 0) {
			count1++;
		}
	}
	int count2 = 0;
	for (int element : a) {
		if (element > 0) {
			count2++;
		}
	}
	int res;
	if (count1 > count2) {
		res = count1 - count2;
	} else {
		res = count2 - count1;
	}
	return res;
}
