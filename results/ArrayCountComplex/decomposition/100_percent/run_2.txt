public static int ArrayCountComplex(int[] a) {
	int count1 = 0;
	for (int element : a) {
		if (element < 0) {
			count1++;
		}
		if (element == 0) {
			count1 = count1;
		} else if (element > 0) {
			count1 = count1;
		} else {
			count1++;
		}
	}
	int count2 = 0;
	for (int element : a) {
		if (element > 0) {
			count2++;
		}
		if (element == 0) {
			count2 = count2;
		} else if (element > 0) {
			count2++;
		} else {
			count2 = count2;
		}
	}
	int result1 = count1 - count2;
	int result2 = 2;
	if (count1 > count2) {
		return result2;
	} else {
		return result2;
	}
}
