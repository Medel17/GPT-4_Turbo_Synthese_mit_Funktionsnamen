public static int ArrayCountComplex(int[] a) {
    int count = 0;
    if (a.length == 0) return 0;
   
    for (int i = 0; i < a.length; i++) {
        if (isComplexNumber(a[i])) {
            count++;
        }
    }
    return count;
}

private static boolean isComplexNumber(int number) {
    return number > 0 && number % 2 != 0;
}
