public static int ArrayCountComplex(int[] a) {
    int count = 0;
    for (int i = 0; i < a.length; i++) {
        boolean isComplex = true;
        for (int j = 2; j <= Math.sqrt(Math.abs(a[i])); j++) {
            if (a[i] % j == 0) {
                isComplex = false;
                break;
            }
        }
        if (isComplex && Math.abs(a[i]) > 1) count++;
    }
    return count;
}
