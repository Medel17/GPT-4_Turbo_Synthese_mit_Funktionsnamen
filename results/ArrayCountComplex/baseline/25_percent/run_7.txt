public static int ArrayCountComplex(int[] a) {
    int count = 0;
    for (int num : a) {
        if (num % 2 == 0 && num > 0) {
            if (hasOnlyEvenDigits(num)) {
                count++;
            }
        }
    }
    return count;
}

private static boolean hasOnlyEvenDigits(int num) {
    while (num > 0) {
        int digit = num % 10;
        if (digit % 2 != 0) {
            return false;
        }
        num /= 10;
    }
    return true;
}
