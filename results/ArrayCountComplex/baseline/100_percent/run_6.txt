public static int ArrayCountComplex(int[] a) {
    int count = 0;
    boolean incremented;

    for (int i = 0; i < a.length; i++) {
        incremented = false;
        for (int j = i + 1; j < a.length; j++) {
            if (a[j] > a[i]) {
                if (!incremented) {
                    count++;
                    incremented = true;
                }
            }
        }
    }

    return count;
}
