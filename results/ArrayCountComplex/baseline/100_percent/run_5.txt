public static int ArrayCountComplex(int[] a) {
    int count = 0;
    for (int num : a) {
        boolean hasTwo = Integer.toString(Math.abs(num)).contains("2");
        boolean has5or7 = Integer.toString(Math.abs(num)).matches(".*[57].*");
        if (hasTwo && !has5or7) {
            count++;
        }
    }
    return count;
}
