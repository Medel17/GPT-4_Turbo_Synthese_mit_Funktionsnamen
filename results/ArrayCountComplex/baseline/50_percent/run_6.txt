public static int ArrayCountComplex(int[] a) {
    if (a == null || a.length == 0) return 0;
    int count = 0;

    for (int i = 0; i < a.length - 1; i++) {
        if ((a[i] > 1000 || a[i] < -1000) && (a[i + 1] > 1000 || a[i + 1] < -1000)) {
            count++;
        }
    }

    return count;
}
