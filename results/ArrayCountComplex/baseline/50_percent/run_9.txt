public static int ArrayCountComplex(int[] a) {
    int countComplex = 0;
    boolean complexLast = false;

    for (int i = 0; i < a.length; i++) {
        boolean isComplex = isNumberComplex(a[i]);
        if (isComplex && (i == 0 || !complexLast)) {
            countComplex++;
        }
        complexLast = isComplex;
    }

    return countComplex;
}

private static boolean isNumberComplex(int number) {
    return (number % 2 != 0 || number < 0) && (number % 5 == 0);
}
