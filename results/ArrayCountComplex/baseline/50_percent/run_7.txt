public static int ArrayCountComplex(int[] a) {
    int count = 0;
    for (int i = 0; i < a.length; i++) {
        if (isComplex(a[i])) {
            count++;
        }
    }
    return count;
}

private static boolean isComplex(int number) {
    // A number is complex if it's absolute value is not a power of two
    int absNumber = Math.abs(number);
    return (absNumber & (absNumber - 1)) != 0;
}
