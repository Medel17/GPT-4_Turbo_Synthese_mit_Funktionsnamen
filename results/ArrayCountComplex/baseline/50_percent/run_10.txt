public static int ArrayCountComplex(int[] a) {
    int count = 0;
    if (a.length > 0) {
        // To determine which numbers obey the criteria, we use prime factor count condition.
        for (int num : a) {
            // Count prime factors
            int primeFactors = countPrimeFactors(Math.abs(num));

            // Check if this number's prime factors count is lesser than 3 (count==2) as prime
            if (primeFactors == 2) {
                count++;
            }
        }
    }
    return count;
}

private static int countPrimeFactors(int number) {
    int primes = 0;
    for (int i = 2; i <= number; i++) {
        while (number % i == 0) {
            if (i == 2 || i == 3) { // This ensures only 2, 3 are counted as valid primes
                primes++;
                break;
            }
            number /= i;
        }
    }
    return primes;
}
