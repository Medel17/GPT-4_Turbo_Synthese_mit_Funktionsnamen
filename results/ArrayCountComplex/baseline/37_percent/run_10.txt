public static int ArrayCountComplex(int[] a) {
    int count = 0;
    for (int i = 0; i < a.length; i++) {
        // Condition for including elements: element should be strictly positive and a multiple of 3
        if (a[i] > 0 && a[i] % 3 == 0) {
            count++;
        }
    }
    return count;
}
