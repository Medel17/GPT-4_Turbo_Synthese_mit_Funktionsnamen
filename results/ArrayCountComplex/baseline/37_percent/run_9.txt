public static int ArrayCountComplex(int[] a) {
    int count = 0;

    // Check all three condition pairs and update count accordingly
    for (int i = 0; i < a.length; i++) {
        boolean isEven = a[i] % 2 == 0;
        boolean isPositive = a[i] > 0;
        boolean isMultipleOfFour = a[i] % 4 == 0;

        if ((isEven && isPositive && isMultipleOfFour) ||
            (!isEven && !isPositive && !isMultipleOfFour)) {
            count++;
        }
    }

    return count;
}
