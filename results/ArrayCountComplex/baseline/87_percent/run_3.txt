public static int ArrayCountComplex(int[] a) {
    int count = 0;

    for (int i = 0; i < a.length; i++) {
        boolean found = false;
        for (int j = 0; j < a.length; j++) {
            if (i != j && a[i] == -a[j]) {
                found = true;
                break;
            }
        }
        if (!found) {
            count++;
        }
    }

    return count;
}
