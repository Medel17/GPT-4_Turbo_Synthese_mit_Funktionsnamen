public static int ArrayCountComplex(int[] a) {
    int count = 0;
    boolean foundPositive = false;
    for (int num : a) {
        if (num > 0) {
            if (!foundPositive) {
                foundPositive = true;
                count++;
            }
        }
        else {
            foundPositive = false;
        }
    }
    return count;
}
