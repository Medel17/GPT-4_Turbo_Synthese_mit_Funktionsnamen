public static int ArrayCountComplex(int[] a) {
    int count = 0;
    for (int i = 0; i < a.length - 2; i++) {
        if ((a[i] % 2 == 1 || a[i] % 2 == -1) && (a[i + 1] % 2 == 1 || a[i + 1] % 2 == -1) && (a[i + 2] % 2 == 1 || a[i + 2] % 2 == -1)) {
            count++;
        }
    }
    return count;
}
