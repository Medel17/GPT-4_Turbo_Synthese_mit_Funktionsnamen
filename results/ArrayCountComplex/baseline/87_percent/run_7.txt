public static int ArrayCountComplex(int[] a) {
    if (a.length == 0) {
        return 0;
    }
    
    int count = 0;
    boolean consecutivePositiveSequence = false;

    for (int i = 0; i < a.length; i++) {
        if (a[i] > 0 && !consecutivePositiveSequence) {
            consecutivePositiveSequence = true;
            count++;
        } else if (a[i] <= 0) {
            consecutivePositiveSequence = false;
        }
    }

    return count;
}
