public static String DoubleLetters(String input) {
	String output = "";
	for (int i = 0; i < input.length(); i++) {
		char c = input.charAt(i);
		int count = 1;
		// ### HOLE BEGIN clamp1
        if (c == 'z' || c == 'm') {
            count = 2;
        }
		// ### HOLE END
		// ### HOLE BEGIN clamp2
        if (c == 'A' || c == 'M') {
            count = 2;
        }
		// ### HOLE END
		// ### HOLE BEGIN clamp3
        if (c == '!') {
            count = 3;
        }
		// ### HOLE END
		while (count > 0) {
			output += c;
			count--;
		}
	}
	return output;
}
