public static String DoubleLetters(String input) {
    String output = "";
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        int count = 1;

        if (c >= 'a' && c <= 'z') {
            count += 1;
        }

        if (c >= 'A' && c <= 'Z') {
            count += 1;
        }

        if (c == '!') {
            count += 2;
        }
        
        while (count > 0) {
            output += c;
            count--;
        }
    }
    return output;
}
