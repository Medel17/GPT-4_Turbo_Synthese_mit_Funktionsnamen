public static String DoubleLetters(String input) {
	String output = "";
	for (int i = 0; i < input.length(); i++) {
		char c = input.charAt(i);
		int count = 2; // Each character needs to duplicated, thus count should start from 2

		while (count > 0) {
			output += c;
			count--;
		}
	}
	return output;
}
