public static String DoubleLetters(String input) {
    String output = "";
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        int count = 1;

        // ### Missing section
        if (Character.isLetter(c) || Character.isDigit(c)) {
            count = 2; // Double the count for letters and digits
        } else if (!Character.isWhitespace(c)) {
            count = 2; // Double the count for non-whitespace and non-alphanumeric characters
        }
        // ### End of Missing section

        while (count > 0) {
            output += c;
            count--;
        }
    }
    return output;
}
