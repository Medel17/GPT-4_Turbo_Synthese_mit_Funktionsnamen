public static String DoubleLetters(String input) {
	String output = "";
	for (int i = 0; i < input.length(); i++) {
		char c = input.charAt(i);
		int count = 1;

		// Check if the character is alphanumeric or a special character which should be doubled
		if (Character.isLetterOrDigit(c) || "!&.'".indexOf(c) >= 0) {
			count = 2;
		}

		while (count > 0) {
			output += c;
			count--;
		}
	}
	return output;
}
