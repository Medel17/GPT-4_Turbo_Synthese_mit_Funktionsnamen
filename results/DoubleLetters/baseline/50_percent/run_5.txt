public static String DoubleLetters(String input) {
    String output = "";
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        int count = 2; // Change the initial count to 2 for doubling

        // Doubling the characters from the input.
        while (count > 0) {
            output += c;
            count--;
        }
    }
    return output;
}
