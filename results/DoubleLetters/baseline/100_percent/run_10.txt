public static String DoubleLetters(String input) {
    String output = "";
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        int count = 1;
        
        if (Character.isLetter(c)) {
            if (Character.isLowerCase(c) || Character.isUpperCase(c)) {
                count = 2;
            }
        } else {
            count = 3;
        }

        while (count > 0) {
            output += c;
            count--;
        }
    }
    return output;
}
