public static String DoubleLetters(String input) {
    String output = "";
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        int count = 1;

        // Determine the number of duplications
        if (Character.isLetter(c)) {
            count = 2; // Double the character if it's a letter
        } else if (Character.isDigit(c)) {
            count = 1; // Leave digits as is
        } else {
            count = 2; // Double other characters (punctuation, symbols)
        }

        while (count > 0) {
            output += c;
            count--;
        }
    }
    return output;
}
