public static String DoubleLetters(String input) {
    String output = "";
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        int count = 1;
        
        if (Character.isLetter(c) || Character.isDigit(c) || "`~!@#$%^&*()_-+=|\\{}[]:;\"'<>,.?/".indexOf(c) >= 0) {
            count = 2;
        }

        while (count > 0) {
            output += c;
            count--;
        }
    }
    return output;
}
