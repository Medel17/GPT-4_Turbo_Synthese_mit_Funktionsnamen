public static int ClampToHundred(int n) {
    int result;
    // ### HOLE BEGIN clamp1
    result = n;
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    if (n > 100) result = 100;
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    if (n < -100) result = -100;
    // ### HOLE END
    return result;
}
