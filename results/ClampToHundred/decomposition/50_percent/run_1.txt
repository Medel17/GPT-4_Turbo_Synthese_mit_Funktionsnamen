public static int ClampToHundred(int n) {
    int result;
    // ### HOLE BEGIN clamp1
    if (n < -100) result = -100;
    else result = n;
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    if (n > 100) result = 100;
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    // No change needed for values between -100 and 100
    // ### HOLE END
    return result;
}
