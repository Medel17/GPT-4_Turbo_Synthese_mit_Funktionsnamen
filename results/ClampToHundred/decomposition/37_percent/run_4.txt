public static int ClampToHundred(int n) {
    int result;
    // ### HOLE BEGIN clamp1
    result = n < -100 ? -100 : n;
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    result = result > 100 ? 100 : result;
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    result = (result < -100 || result > 100) ? result : n;
    // ### HOLE END
    return result;
}
