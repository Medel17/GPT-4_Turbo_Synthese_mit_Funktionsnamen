public static int ClampToHundred(int n) {
    int result;
    // ### HOLE BEGIN clamp1
    if (n < -100) {
        result = -100;
    } else {
        result = n;
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    if (n > 100) {
        result = 100;
    } else {
        result = n;
    }
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    result = n;
    // ### HOLE END
    return result;
}
