public static int ClampToHundred(int n) {
    int result;
    // ### HOLE BEGIN clamp1
    if (n < -100) {
    	result = -100;
    } else {
	    result = n;
	}
	// ### HOLE END
	// ### HOLE BEGIN clamp2
    if (result > 100) {
    	result = 100;
    }
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	// No change needed, result is already set correctly from previous conditions.
	// ### HOLE END
	return result;
}
