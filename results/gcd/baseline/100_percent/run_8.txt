public static int gcd(int a, int b) {
    // Ensure the values are positive since gcd should be positive or zero
    a = Math.abs(a);
    b = Math.abs(b);

    // Base termination condition
    if (b == 0) {
        return a;
    }

    // Recursive call for Euclidean algorithm
    return gcd(b, a % b);
}
