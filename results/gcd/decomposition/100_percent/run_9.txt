public static int gcd(int a, int b) {
    // ### HOLE BEGIN clamp1
    a = Math.abs(a);
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    b = Math.abs(b);
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    if (a == 0 && b == 0) return 0;
    else if (a == 0) return b;
    else if (b == 0) return a;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    while (b != 0) {
        // ### HOLE BEGIN clamp5
        int temp = a % b;
        // ### HOLE END
        
        // ### HOLE BEGIN clamp6
        a = b;
        // ### HOLE END
        
        // Update b to the modulus result for the next iteration
        b = temp;
    }
    // ### HOLE END
    return a;
}
