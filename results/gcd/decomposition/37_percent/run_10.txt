public static int gcd(int a, int b) {
    if (a < 0) a = -a;
    if (b < 0) b = -b;
    if (a == 0 && b == 0) return 0;
    if (a == 0) return b;

    while (b != 0) {
        if (a > b) {
            a = a - b;
        } else {
            int temp = b - a;
            b = temp;
        }
        if (b != 0) {
            int temp = a;
            a = b;
            b = temp;
        }
    }
    return a;
}
