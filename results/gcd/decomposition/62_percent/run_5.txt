public static int gcd(int a, int b) {
	// ### HOLE BEGIN clamp1
	if (a == 0) return 0; else return a;
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	if (b == 0) return 0; else return b;
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	if (a == 0) return b;
	if (b == 0) return a;
	// ### HOLE END
	// ### HOLE BEGIN clamp4
	while (b != 0) {
		// ### HOLE BEGIN clamp5
		int temp = a % b;
		// ### HOLE END		
		// ### HOLE BEGIN clamp6
		a = b;
		// ### HOLE END
		b = temp;
	}
	// ### HOLE END
	return a;
}
