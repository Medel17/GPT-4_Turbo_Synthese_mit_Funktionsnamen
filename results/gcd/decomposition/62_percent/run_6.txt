public static int gcd(int a, int b) {
    // ### HOLE BEGIN clamp1
    if (a < 0) a = -a;
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    if (b < 0) b = -b;
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    if (a == 0) return b;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    while (b != 0) {
        // ### HOLE BEGIN clamp5
        int temp = b;
        b = a % b;
        // ### HOLE END
        
        // ### HOLE BEGIN clamp6
        a = temp;
        // ### HOLE END
    }
    // ### HOLE END
    return a;
}
