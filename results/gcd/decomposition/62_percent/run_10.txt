public static int gcd(int a, int b) {
    // ### HOLE BEGIN clamp1
    if (a == 0) return a;
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    if (b == 0) return b;
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    if (a == 0) return b;
    if (b == 0) return a;
    // ### HOLE END
    // ### HOLE BEGIN clamp4
    while (a != b) {
        // ### HOLE BEGIN clamp5
        if (a > b) {
            a = a - b;
        // ### HOLE END
        
        // ### HOLE BEGIN clamp6
        } else {
            int temp = a;
            a = b;
            b = temp % b;
        }
        // ### HOLE END
    }
    // ### HOLE END
    return a;
}
