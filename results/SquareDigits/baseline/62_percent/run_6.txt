public static String SquareDigits(int x) {
    if (x < 0) return "";
    String result = "";
    while (x > 0) {
        int d = x % 10; // get the last digit
        result = (d * d) + result; // square it, and prepend it to result (to maintain order)
        x = x / 10; // reduce x by removing last digit
    }
    return result;
}
