public static String SquareDigits(int x) {
    if (x < 0) return ""; // Return empty string if the number is negative

    String result = "";
    while (x > 0) {
        int digit = x % 10; // Extract the last digit
        result = (digit * digit) + result; // Square the digit and prepend it to the result
        x /= 10; // Remove the last digit
    }
    return result;
}
