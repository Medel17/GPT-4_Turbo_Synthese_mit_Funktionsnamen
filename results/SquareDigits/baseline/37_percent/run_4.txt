public static String SquareDigits(int x) {
    String result = "";
    if (x < 0) return result; // this handles the case where input is negative
    while (x > 0) {
        int digit = x % 10; // get the last digit of the number
        result = (digit * digit) + result; // square the digit and prepend to result (to maintain order)
        x = x / 10; // remove the last digit
    }
    return result;
}
