public static String SquareDigits(int x) {
    if (x < 0) return ""; // Ignore negative input
    String result = "";
    while (x > 0) {
        int digit = x % 10; // Get the last digit
        result = (digit * digit) + result; // Square the digit and prepend to the result
        x = x / 10; // Move to the next digit
    }
    return result;
}
