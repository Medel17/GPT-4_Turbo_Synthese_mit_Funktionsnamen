public static String SquareDigits(int x) {
	String result = "";
	// if the number is negative, return an empty string
	if (x < 0) {
		return result;
	}
	while (x > 0) {
		int digit = x % 10; // get the last digit
		result = (digit * digit) + result; // square of digit and prepend to the result
		x /= 10; // remove the last digit
	}
	return result;
}
