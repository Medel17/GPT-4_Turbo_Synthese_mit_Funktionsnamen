public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b; // 'b' is middle between 'a' and 'c'
        } else if (a < c) {
            return c; // 'c' is middle when 'b' > 'c' > 'a'
        } else {
            return a; // 'a' falls between 'c' and 'b'
        }
    } else {
        if (a < c) {
            return a; // 'a' is middle between 'b' and 'c'
        } else if (b < c) {
            return c; // 'c' is middle when 'a' > 'c' > 'b'
        } else {
            return b; // 'b' falls between 'c' and 'a'
        }
    }
}
