public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b; // b is in the middle
        } else if (a < c) {
            return c; // c is in the middle
        } else {
            return a; // a is in the middle
        }
    } else {
        if (a < c) {
            return a; // a is in the middle
        } else if (b < c) {
            return c; // c is in the middle
        } else {
            return b; // b is in the middle
        }
    }
}
