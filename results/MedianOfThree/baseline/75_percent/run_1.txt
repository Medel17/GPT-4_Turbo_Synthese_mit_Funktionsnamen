public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (a >= c) {
            return a;
        } else if (c < b) {
            return c;
        } else {
            return b;
        }
    } else {
        if (b >= c) {
            return b;
        } else if (c < a) {
            return c;
        } else {
            return a;
        }
    }
}
