public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b;  // b is median of a < b < c
        } else if (a < c) {
            return c;  // c is median of a < c <= b
        } else {
            return a;  // a is median of c <= a < b
        }	    
    } else {
        if (a < c) {
            return a;  // a is the median of b <= a < c
        } else if (b < c) {
            return c;  // c is the median of b <= c <= a
        } else {
            return b;  // b is the median of c <= b <= a
        }
    }
}
