public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b; // b is median
        } else if (a < c) {
            return c; // c is median
        } else {
            return a; // a is median
        }
    } else {
        if (a < c) {
            return a; // a is median
        } else if (b < c) {
            return c; // c is median
        } else {
            return b; // b is median
        }
    }
}
