public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b;
        } else { // c <= b
            if (a < c) {
                return c;
            } else {
                return a;
            }
        }
    } else { // a >= b
        if (a < c) {
            return a;
        } else { // c <= a
            if (b < c) {
                return c;
            } else {
                return b;
            }
        }
    }
}
