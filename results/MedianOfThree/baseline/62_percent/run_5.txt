public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b; // a < b < c -> median is b
        } else {
            return Math.max(a, c); // a < b and b >= c -> median is max(a, c)
        }
    } else { // a >= b
        if (a < c) {
            return a; // b <= a < c -> median is a
        } else {
            return Math.max(b, c); // b <= a and a >= c -> median is max(b, c)
        }
    }
}
