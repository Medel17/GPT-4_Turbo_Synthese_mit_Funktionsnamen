public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b;
        } else {
            return Math.max(a, c);
        }
    } else {
        if (a < c) {
            return a;
        } else {
            return Math.max(b, c);
        }
    }
}
