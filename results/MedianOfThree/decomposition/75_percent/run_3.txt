public static int MedianOfThree(int a, int b, int c) {
    if (a < b) {
        if (b < c) {
            return b;
        } else {
            return (a > c) ? a : c;
        }
    } else {
        if (a < c) {
            return a;
        } else {
            return (b > c) ? b : c;
        }
    }
}
