public static boolean LeapYear(int y) {
	// ### HOLE BEGIN clamp1
	if (y < 0) return false;
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	if (y % 4 == 0 && y >= 0) return true;
	// ### HOLE END
	// ### HOLE BEGIN clamp3
	if (y % 100 == 0 && y % 400 != 0) return false;
	// ### HOLE END
	return y % 4 == 0;
}
