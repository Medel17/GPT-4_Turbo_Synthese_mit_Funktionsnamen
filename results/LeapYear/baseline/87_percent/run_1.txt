public static boolean LeapYear(int y) {
	if (y % 4 != 0) {
		return false; // not divisible by 4
	} else if (y % 100 != 0) {
		return true; // divisible by 4 but not 100
	} else if (y % 400 == 0) {
		return true; // divisible by 100 and 400
	} else {
		return false; // divisible by 100 but not 400
	}
}
