public static String Syllables(String input) {
    int count = 0;
    for (int i = 0; i < input.length(); i++) {
        // ### HOLE BEGIN clamp1
        char c = input.charAt(i);
        // ### HOLE END
        // ### HOLE BEGIN clamp2
        if ("aeiouyAEIOUY".indexOf(c) != -1) {
            count++;
        }
        // ### HOLE END
    }
    // ### HOLE BEGIN clamp3
    return Integer.toString(count);
    // ### HOLE END
}
