public static String Syllables(String input) {
    int count = 0;
    for (int i = 0; i < input.length(); i++) {
        char c = input.charAt(i);
        switch (c) {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':
                // This is to avoid counting the same vowel cluster multiple times
                if (i == 0 || (i > 0 && !isVowel(input.charAt(i - 1)))) {
                    count++;
                }
                break;
            default:
                // Non-vowel characters do not affect the count
                break;
        }
    }
    return Integer.toString(count);
}

private static boolean isVowel(char c) {
    return "aeiou".indexOf(c) >= 0;
}
