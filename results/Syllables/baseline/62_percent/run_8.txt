public static String Syllables(String input) {
    int count = 0;
    for (int i = 0; i < input.length(); i++) {
        char ch = input.charAt(i);
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' || ch == 'y') {
            // if it is not the first vowel in a sequence
            if (i == 0 || (input.charAt(i - 1) != 'a' && input.charAt(i - 1) != 'e' && input.charAt(i - 1) != 'i' &&
                           input.charAt(i - 1) != 'o' && input.charAt(i - 1) != 'u' && input.charAt(i - 1) != 'y')) {
                count++;
            }
        }
    }
    return String.valueOf(count);
}
