public static int pairedDigits(String input) {
    int sum = 0;
    // Check if input string length is eligible for digit pairing
    if (input == null || input.length() < 2) {
        return sum;
    }
    
    // Loop through all characters in the string minus the last character
    for (int i = 0; i < input.length() - 1; i++) {
        // Check if current character and the next character are the same
        if (input.charAt(i) == input.charAt(i+1)) {
            // Convert the character to integer and add it to the sum
            sum += Integer.parseInt(Character.toString(input.charAt(i)));
        }
    }
    return sum;
}
