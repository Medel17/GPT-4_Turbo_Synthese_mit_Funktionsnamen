public static int pairedDigits(String input) {
    int sum = 0;
    // checking if input is not null and has even length
    if (input == null || input.length() % 2 != 0) {
        return sum;
    }
    
    // iterate through the string by 2 steps
    for (int i = 0; i < input.length(); i+=2) {
        // check if current and next characters are the same
        if (input.charAt(i) == input.charAt(i + 1)) {
            // convert the character to its numeric value and add to sum
            sum += Integer.parseInt(String.valueOf(input.charAt(i)));
        }
    }
    
    return sum;
}
