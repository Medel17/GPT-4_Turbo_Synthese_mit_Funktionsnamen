public static int pairedDigits(String input) {
    int sum = 0;
    if (input == null || input.length() <= 1) {
        return sum;
    }
    // Traverse the input to find pairs of sequential digits where both digits are odd or even
    for (int i = 0; i < input.length() - 1; i++) {
        char currentChar = input.charAt(i);
        char nextChar = input.charAt(i + 1);
        // Check if both characters are digits
        if (Character.isDigit(currentChar) && Character.isDigit(nextChar)) {
            int currentDigit = Character.getNumericValue(currentChar);
            int nextDigit = Character.getNumericValue(nextChar);
            // Check if both digits are odd or both are even
            if ((currentDigit % 2 == nextDigit % 2)) {
                sum += 1;
            }
        }
    }
    return sum;
}
