public static int pairedDigits(String input) {
    int sum = 0;
    
    if (input == null || input.isEmpty() || input.length() == 1) {
        // If input is null, empty, or has only one character, return 0 since no pairs can be formed.
        return sum;
    }

    // Iterate through the string checking subsequent characters
    for (int i = 0; i < input.length() - 1; i++) {
        // Convert current character and the next character to their numeric values
        int currentDigit = Character.getNumericValue(input.charAt(i));
        int nextDigit = Character.getNumericValue(input.charAt(i+1));
        
        // Check if two adjacent digits are the same and that they meet the condition of the repeated digit
        if (currentDigit == nextDigit) {
            // If two consecutive characters are the same, add that digit to the sum.
            sum += currentDigit;
        }
    }
    
    return sum;
}
