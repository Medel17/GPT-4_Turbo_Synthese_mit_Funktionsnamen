public static int pairedDigits(String input) {
		int sum = 0;

		// Validate input and handle edge cases
		if (input == null || input.length() < 2) {
			return sum;
		}

		// Iterate through the string to detect consecutive paired digits
		for (int i = 0; i < input.length() - 1; i++) {
			char currentChar = input.charAt(i);
			char nextChar = input.charAt(i + 1);
			
			// Check if the current and next characters are digits
			if (Character.isDigit(currentChar) && Character.isDigit(nextChar)) {
				// Check if characters are the same
				if (currentChar == nextChar) {
					// Add the numerical value of the digit to the sum
					sum += Character.getNumericValue(currentChar);
				}
			}
		}
		return sum;
}
