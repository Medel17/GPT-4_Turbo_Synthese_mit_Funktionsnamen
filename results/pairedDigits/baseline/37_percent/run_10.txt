public static int pairedDigits(String input) {
    int sum = 0;
    // ### HOLE BEGIN precondition
    if (input == null || input.length() <= 1) {
        return 0;
    }

    // Iterate through the string
    for (int i = 0; i < input.length() - 1; i++) {
        // Check if current and next characters are digits
        if (Character.isDigit(input.charAt(i)) && Character.isDigit(input.charAt(i + 1))) {
            // Calculate the paired sum
            int currentDigit = Character.getNumericValue(input.charAt(i));
            int nextDigit = Character.getNumericValue(input.charAt(i + 1));
            if (currentDigit == nextDigit) {
                sum += currentDigit + nextDigit;
            }
        }
    }
    // ### End of Missing section
    return sum;
}
