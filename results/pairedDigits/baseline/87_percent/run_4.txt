public static int pairedDigits(String input) {
    int sum = 0;
    // Check if the string is empty or has less than 2 characters, directly return 0
    if (input == null || input.length() < 2) {
        return 0;
    }

    // Loop through the string and check adjacent characters
    for (int i = 0; i < input.length() - 1; i++) {
        if (input.charAt(i) == input.charAt(i + 1)) {
            // Convert the character to its numerical value and add it to the sum
            sum += Character.getNumericValue(input.charAt(i));
        }
    }

    // Return the accumulated sum of the digits that are paired (adjacent and the same)
    return sum;
}
