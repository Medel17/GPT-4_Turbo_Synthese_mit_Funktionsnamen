public static int pairedDigits(String input) {
    int sum = 0;
    if (input == null || input.length() < 2) {
        return sum;
    }
    char[] chars = input.toCharArray();
    for (int i = 0; i < chars.length - 1; i++) {
        if (chars[i] == chars[i + 1]) {
            sum += Character.getNumericValue(chars[i]);
        }
    }
    return sum;
}
