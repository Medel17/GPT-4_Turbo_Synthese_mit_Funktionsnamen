public static int pairedDigits(String input) {
    int sum = 0;
    int n = input.length();

    for (int i = 0; i < n - 1; i++) {
        if (input.charAt(i) == input.charAt(i + 1)) {
            sum += Character.getNumericValue(input.charAt(i));
        }
    }

    if (n > 1 && input.charAt(0) == input.charAt(n - 1)) {
        sum += Character.getNumericValue(input.charAt(0));
    }

    return sum;
}
