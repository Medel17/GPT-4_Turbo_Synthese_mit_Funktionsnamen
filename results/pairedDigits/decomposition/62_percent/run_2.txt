public static int pairedDigits(String input) {
    int sum = 0;
    for (int i = 0; i < input.length() - 1; i++) {
        char currentChar = input.charAt(i);
        char nextChar = input.charAt(i + 1);
        if (currentChar == nextChar) {
            sum += Character.getNumericValue(currentChar);
        }
    }
    return sum;
}
