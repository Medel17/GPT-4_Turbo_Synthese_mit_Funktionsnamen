public static boolean CompareStringLengths(String str1, String str2, String str3) {
	// ### HOLE BEGIN clamp1
	boolean isStr1Str2SameLength = str1.length() == str2.length();
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	boolean isStr2Str3SameLength = str2.length() == str3.length();
	// ### HOLE END
	return !isStr1Str2SameLength && !isStr2Str3SameLength;
}
