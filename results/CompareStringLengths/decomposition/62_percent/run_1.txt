public static boolean CompareStringLengths(String str1, String str2, String str3) {
	// ### HOLE BEGIN clamp1
	boolean equalLength1and2 = str1.length() == str2.length();
	// ### HOLE END
	// ### HOLE BEGIN clamp2
	boolean equalLength2and3 = str2.length() == str3.length();
	// ### HOLE END
	return equalLength1and2 && equalLength2and3;
}
