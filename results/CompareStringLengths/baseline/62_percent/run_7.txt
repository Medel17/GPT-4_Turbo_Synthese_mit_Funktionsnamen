public static boolean CompareStringLengths(String str1, String str2, String str3) {
    // Get the lengths of each string
    int len1 = str1.length();
    int len2 = str2.length();
    int len3 = str3.length();

    // Check if all lengths are different from each other
    if (len1 != len2 && len2 != len3 && len1 != len3) {
        return true;
    } else {
        return false;
    }
}
