public static int MultipleComparisons(int a1, int a2, int b1, int b2) {
    int result;
    if (a1 > b1) {
        result = a1 - b1;
    } else {
        result = b1 - a1;
    }
    
    if (a2 > b2) {
        result += a2 - b2;
    } else {
        result += b2 - a2;
    }
    
    return result;
}
