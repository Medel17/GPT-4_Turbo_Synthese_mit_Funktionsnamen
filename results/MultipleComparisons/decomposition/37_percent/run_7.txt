public static int MultipleComparisons(int a1, int a2, int b1, int b2) {
    // ### HOLE BEGIN clamp1
    int GA = (a1 > a2) ? a1 : a2;
    // ### HOLE END
    // ### HOLE BEGIN clamp2
    int SB = (b1 < b2) ? b1 : b2;
    // ### HOLE END
    // ### HOLE BEGIN clamp3
    int result = (GA > -SB) ? GA : -SB;
    // ### HOLE END

    return result;
}
