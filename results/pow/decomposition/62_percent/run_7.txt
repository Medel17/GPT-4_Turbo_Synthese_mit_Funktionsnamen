public static int pow(int x, int n) {
    if (x == 0 && n <= 0) { // precondition
        return 0;
    }
    int k = n;
    int p = x;
    int y = 1;
    while (k > 0) {
        if (k % 2 == 0) {
            p = p * p; // p squared
            k = k / 2; // k half
        } else {
            y = y * p; // y times p
            k = k - 1; // k decrement
        }
    }
    return y;
}
