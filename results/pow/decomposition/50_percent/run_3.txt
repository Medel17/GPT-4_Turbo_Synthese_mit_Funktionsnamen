public static int pow(int x, int n) {
    if (n < 0 || (x == 0 && n == 0)) return 0; // Precondition check for invalid inputs
    int k = n;
    int p = x;
    int y = 1;
    while (k > 0) {
        if (k % 2 == 0) {
            p = p * p; // Squaring p
            k = k / 2; // Halving k
        } else {
            y = y * p; // Multiplying y by p
            k = k - 1; // Decreasing k by 1
        }
    }
    return y;
}
