public static int pow(int x, int n) {
    if (x == 0 && n == 0) return 0; // Handles hole 1
    int k = n;
    int p = x;
    int y = 1;
    while (k > 0) {
        if (k % 2 == 0) {
            p = p * p; // Handles hole 2
            k = k / 2; // Handles hole 3
        } else {
            y = y * p; // Handles hole 4
            k = k - 1; // Handles hole 5
        }
    }
    return y;
}
