public static int pow(int x, int n) {
    if (x == 0 && n == 0) return 0; // handles 0^0 as special case
    int k = n;
    int p = x;
    int y = 1;
    while (k > 0) {
        if (k % 2 == 0) {
            p = p * p; // pSquared
            k = k / 2; // kHalf
        } else {
            y = y * p; // yTimes
            k = k - 1; // kDec
        }
    }
    return y;
}
