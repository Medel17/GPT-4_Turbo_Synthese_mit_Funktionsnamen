public static int pow(int x, int n) {
    // Handle base cases and constraints, and input validation
    if (n < 0) {
        return 0;
    }
    if (n == 0) {
        return 1;
    }
    if (x == 0) {
        return 0;
    }

    int y = 1;
    for (int i = 0; i < n; i++) {
        y *= x;
    }
    return y;
}
