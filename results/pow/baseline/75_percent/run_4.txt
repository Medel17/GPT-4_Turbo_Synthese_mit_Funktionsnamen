public static int pow(int x, int n) {
    // Handle invalid exponent cases
    if (n < 0) {
        return 0;
    }
    // Initial case where exponent is zero
    if (n == 0) {
        return 1;
    }
    // Calculate power for n > 0
    int y = 1;
    for(int i = 0; i < n; i++) {
        y *= x;
    }
    return y;
}
