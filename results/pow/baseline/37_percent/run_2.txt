public static int pow(int x, int n) {
    // Checking input constraints
	if (n < 0) throw new IllegalArgumentException("Exponent must be non-negative");

	int y = 1;
    // Implementing exponentiation loop
	while (n > 0) {
		if (n % 2 == 1) {
			y *= x;
			n -= 1;
		}
		x *= x;
		n /= 2;
	}
	return y;
}
