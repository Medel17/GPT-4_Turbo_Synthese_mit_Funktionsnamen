public static int pow(int x, int n) {
    // ### Missing section
    if (n < 0) {
        throw new IllegalArgumentException("Exponent must be non-negative.");
    }

    int y = 1;
    while (n > 0) {
        if (n % 2 == 1) {
            y = y * x;
            n--;
        } else {
            x = x * x;
            n /= 2;
        }
    }
    // ### End of Missing section
    return y;
}
