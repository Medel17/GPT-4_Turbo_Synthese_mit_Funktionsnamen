public static int pow(int x, int n) {
    // check for n less than 0 as we don't handle negative exponents for simplicity:
    // also handle base cases
    if (n < 0) {
        return 0; // Typically you'd want fractional results, but returning 0 for simplification
    } else if (n == 0) {
        return 1; // any number to the power of 0 is 1
    } else if (x == 0) {
        return 0; // 0 to any positive power is 0
    }

    int y = 1;
    for (int i = 1; i <= n; i++) {
        y *= x;
    }
    return y;
}
