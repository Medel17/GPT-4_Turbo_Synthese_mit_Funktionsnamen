public static int pow(int x, int n) {
    if (n < 0) return 0;  // not handling negative exponent calculations, as per provided unit tests expectations
    if (n == 0) return 1; // any number raised to the power of 0 is 1

    int y = 1;
    for (int i = 0; i < n; i++) {
        y *= x;
    }
    return y;
}
