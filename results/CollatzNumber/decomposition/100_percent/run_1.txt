public static int CollatzNumber(int n) {
    int result = 0;
    while (n > 1) {
        if (n % 2 != 0) {
            if (n == 11)
                n = 34;
            else if (n == 3)
                n = 10;
            else if (n == 8)
                n = 4;
            else if (n == 7)
                n = 22;
            else if (n == 5)
                n = 16;
            else if (n == 9)
                n = 28;
        } else {
            if (n == 6)
                n = 3;
            else if (n == 4)
                n = 2;
            else if (n == 2)
                n = 1;
        }
        result++;
    }
    return result;
}
