public static int CollatzNumber(int n) {
    int result = 0;
    while (n > 1) {
        // ### HOLE BEGIN clamp1
        if (n % 2 == 0) {
            // ### HOLE BEGIN even clamp2
            n = n / 2;
            // ### HOLE END clamp2
        } else {
            // ### HOLE BEGIN clamp3
            n = 3 * n + 1;
            // ### HOLE END clamp3
        }
        // ### HOLE END clamp1
        result++;
    }
    return result;
}
