public static int CollatzNumber(int n) {
    int result = 0;
    while (n > 1) {
        if (n % 2 != 0) {
            if (n == 11) {
                n = 34;
            } else if (n == 9) {
                n = 28;
            } else if (n == 7) {
                n = 22;
            } else {
                n = 3 * n + 1;
            }
        } else {
            if (n == 4 || n == 2) {
                n = n / 2;
            } else {
                n = n / 2;
            }
        }
        result++;
    }
    return result;
}
