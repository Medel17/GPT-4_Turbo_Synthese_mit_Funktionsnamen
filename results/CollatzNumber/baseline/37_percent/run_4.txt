public static int CollatzNumber(int n) {
    if (n <= 0) return 0;  // Immediate return for non-positive values to handle the negative case

    int result = 0;
    while (n > 1) {
        if (n % 2 == 0) {
            n /= 2;
        } else {
            n = 3 * n + 1;
        }
        result++;
    }
    return result;
}
