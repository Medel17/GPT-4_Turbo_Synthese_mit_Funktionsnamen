public static int CollatzNumber(int n) {
    int result = 0;
    if (n <= 0) return 0;  // Handling non-positive inputs according to the given test
    while (n > 1) {
        if (n % 2 == 0) {
            n = n / 2;
        } else {
            n = 3 * n + 1;
        }
        result++;
    }
    return result;
}
