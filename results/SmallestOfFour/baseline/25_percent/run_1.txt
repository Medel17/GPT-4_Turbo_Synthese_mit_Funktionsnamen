public static int SmallestOfFour(int a, int b, int c, int d) {
	if (a < b) {
        if (a < c && a < d) {
            return a;
        } else if (c < d) {
            return c;
        } else {
            return d;
        }
	} else {
        if (b < c && b < d) {
            return b;
        } else if (c < d) {
            return c;
        } else {
            return d;
        }
    }
}
