public static int SmallestOfFour(int a, int b, int c, int d) {
    if (a < b) {
        // ### HOLE BEGIN clamp1
        int e = Math.min(c, d);
        return Math.min(a, e);
        // ### HOLE END
    } else {
        // ### HOLE BEGIN clamp2
        int f = Math.min(c, d);
        return Math.min(b, f);
        // ### HOLE END
    }
}
