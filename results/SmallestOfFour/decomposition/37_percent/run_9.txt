public static int SmallestOfFour(int a, int b, int c, int d) {
	if (a < b) {
		// ### HOLE BEGIN clamp1
		if (a < c) {
			if (a < d) {
				return a;
			} else {
				return d;
			}
		} else {
			if (c < d) {
				return c;
			} else {
				return d;
			}
		}
		// ### HOLE END
	} else {
		// ### HOLE BEGIN clamp2
		if (b < c) {
			if (b < d) {
				return b;
			} else {
				return d;
			}
		} else {
			if (c < d) {
				return c;
			} else {
				return d;
			}
		}
		// ### HOLE END
	}
}
