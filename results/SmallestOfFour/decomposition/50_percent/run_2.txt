public static int SmallestOfFour(int a, int b, int c, int d) {
    if (a < b) {
        // ### HOLE BEGIN clamp1
        return (c < d) ? Math.min(a, c) : Math.min(a, d);
        // ### HOLE END
    } else {
        // ### HOLE BEGIN clamp2
        return (c < d) ? Math.min(b, c) : Math.min(b, d);
        // ### HOLE END
    }
}
