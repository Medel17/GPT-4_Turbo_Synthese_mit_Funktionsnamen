public static int SmallestOfFour(int a, int b, int c, int d) {
    if (a < b) {
        // ### HOLE BEGIN clamp1
        return Math.min(Math.min(a, c), d);
        // ### HOLE END
    } else {
        // ### HOLE BEGIN clamp2
        return Math.min(Math.min(b, c), d);
        // ### HOLE END
    }
}
