public static String SmallOrLarge(int n) {
		// ### HOLE BEGIN clamp1
		if (n < 1000) {
		    return "small";
		}
		// ### HOLE END
		// ### HOLE BEGIN clamp2
		else if (n >= 2000) {
		    return "large";
		}
		// ### HOLE END
		// ### HOLE BEGIN clamp3
		else {
		    return "";
		}
		// ### HOLE END
	}
