public static double VectorAverage_Double(double[] arr) {
	// ### Missing section

	double sum = 0;
	for (double element : arr) {
		sum += element;
	}
	double length = arr.length;
	if (length == 0) {
		throw new IllegalArgumentException("Array must not be empty");
	}
	return sum / length;
    
	// ### End of Missing section
}
