public static int VectorAverage_Int(int[] arr) {
    if (arr == null || arr.length == 0) {
        return 0; // Return 0 for empty array as handling measure, might be changed depending on specifications
    }

    int sum = 0;
    for (int element : arr) {
        sum += element; // Accumulate the sum of all elements in an array
    }

    return sum / arr.length; // Return the integer average (by integer division)
}
