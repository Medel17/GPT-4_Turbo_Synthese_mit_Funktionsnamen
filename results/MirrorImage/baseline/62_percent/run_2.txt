public static boolean MirrorImage(int[] arr1, int[] arr2) {
	// First check if the two arrays have the same length
	if (arr1.length != arr2.length) {
		return false;
	}

	// Compare elements in arr1 to reversed elements in arr2
	for (int i = 0; i < arr1.length; i++) {
		if (arr1[i] != arr2[arr2.length - 1 - i]) {
			return false; // Encounter a mismatched pair
		}
	}
    
	return true; // All elements match in a mirrored fashion
}
