public static boolean MirrorImage(int[] arr1, int[] arr2) {
    if (arr1.length != arr2.length) return false; // Arrays must be of same length to be mirror images

    for (int i = 0; i < arr1.length; i++) {
        if (arr1[i] != arr2[arr2.length - 1 - i]) {
            return false; // Check if elements are mirror images
        }
    }

    return true;
}
