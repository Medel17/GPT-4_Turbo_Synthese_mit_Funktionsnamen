public static boolean MirrorImage(int[] arr1, int[] arr2) {
    if (arr1.length != arr2.length) {
        return false;
    }

    int left = 0;
    int right = arr2.length - 1;
    while (left < arr1.length) {
        if (arr1[left] != arr2[right]) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}
