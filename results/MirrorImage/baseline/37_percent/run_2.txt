public static boolean MirrorImage(int[] arr1, int[] arr2) {
    // Check if lengths of both arrays are not the same
    if (arr1.length != arr2.length) {
        return false;
    }

    // Compare elements from end of arr2 to start of arr1
    for (int i = 0; i < arr1.length; i++) {
        if (arr1[i] != arr2[arr2.length - 1 - i]) {
            return false;
        }
    }

    return true;
}
