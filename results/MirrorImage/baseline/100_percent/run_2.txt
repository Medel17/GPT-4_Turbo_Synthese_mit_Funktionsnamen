public static boolean MirrorImage(int[] arr1, int[] arr2) {
    if (arr1.length != arr2.length) { 
        return false;
    }
    
    int lastIndex = arr1.length - 1;
    for (int i = 0; i < arr1.length; i++) {
        if (arr1[i] != arr2[lastIndex - i]) {
            return false;
        }
    }
    
    return true;
}
