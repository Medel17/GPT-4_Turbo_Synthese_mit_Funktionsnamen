public static boolean MirrorImage(int[] arr1, int[] arr2) {
    // ### HOLE BEGIN clamp1
    if (arr1.length != arr2.length)
        return false;
    // ### HOLE END
    for (int i = 0; i < arr1.length; i++) {
        // ### HOLE BEGIN clamp2
        int oppositeIndex = arr2.length - 1 - i;
        // ### HOLE END
        // ### HOLE BEGIN clamp3
        if (arr1[i] != arr2[oppositeIndex])
            return false;
        // ### HOLE END
    }
    return true;
}
